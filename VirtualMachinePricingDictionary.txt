<#
.SYNOPSIS
    Monthly Linux VM SKU pricing refresh — dynamically discovers all unique Linux VM
    SKUs in use across the tenant, fetches West Europe Pay-As-You-Go hourly pricing
    from the Azure Retail Prices API, and persists results to dbo.AzureVmSkuPricing.

.DESCRIPTION
    Runs on the 1st of every month at 00:30 UTC. Replaces the static SKU list with
    a live inventory scan across all enabled subscriptions in the tenant. Only Linux
    VM SKUs are priced — Windows OS licensing cost is explicitly excluded.

.NOTES
    Version : 2.0
    Owner   : Cloud Operations Team
    Schedule: 1st of every month at 00:30 UTC
#>

#region ── Configuration ────────────────────────────────────────────────────

$SQL_Server_Name = Get-AutomationVariable     -Name "SQL_Server_Name"
$SQL_DB_Name     = Get-AutomationVariable     -Name "SQL_DB_Name"
$SQLServerCred   = Get-AutomationPSCredential -Name "SQLServerCred"

$PricingApiUrl = "https://prices.azure.com/api/retail/prices"
$ApiVersion    = "2023-01-01-preview"
$Region        = "westeurope"
$OSType        = "Linux"

#endregion

#region ── Helper: Fetch Linux-only price for a single SKU ──────────────────

function Get-LinuxVmSkuPrice {
    param (
        [string]$SkuName,
        [string]$Region,
        [string]$ApiUrl,
        [string]$ApiVersion
    )

    $Filter = "armSkuName eq '$SkuName' " +
              "and armRegionName eq '$Region' " +
              "and priceType eq 'Consumption' " +
              "and serviceName eq 'Virtual Machines' " +
              "and not contains(meterName, 'Windows') " +
              "and not contains(skuName, 'Windows')"

    $Uri = "$ApiUrl`?api-version=$ApiVersion&`$filter=$([Uri]::EscapeDataString($Filter))"

    try {
        $Response = Invoke-RestMethod -Uri $Uri -Method GET -ContentType "application/json"

        if ($Response.Items -and $Response.Items.Count -gt 0) {
            $LinuxItem = $Response.Items | Where-Object {
                $_.skuName       -notmatch "Windows|Low Priority|Spot" -and
                $_.meterName     -notmatch "Windows|Low Priority|Spot" -and
                $_.unitOfMeasure -eq "1 Hour"
            } | Select-Object -First 1

            if ($LinuxItem) {
                return [PSCustomObject]@{
                    VmSkuName    = $SkuName
                    PricePerHour = [decimal]$LinuxItem.retailPrice
                    MeterName    = $LinuxItem.meterName
                    SkuNameRaw   = $LinuxItem.skuName
                    Found        = $true
                }
            }
            Write-Warning "  Results returned for '$SkuName' but none matched Linux/standard filter."
        }
        else {
            Write-Warning "  No API results for SKU: $SkuName (may not be available in $Region)"
        }
    }
    catch {
        Write-Warning "  API call failed for '$SkuName': $_"
    }

    return [PSCustomObject]@{
        VmSkuName    = $SkuName
        PricePerHour = 0
        MeterName    = $null
        SkuNameRaw   = $null
        Found        = $false
    }
}

#endregion

#region ── Main ─────────────────────────────────────────────────────────────

Write-Output "========================================"
Write-Output "VM SKU Pricing Refresh ($OSType Only)"
Write-Output "Timestamp : $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') UTC"
Write-Output "Region    : $Region"
Write-Output "========================================"

# ── Step 0: Discover all unique Linux VM SKUs across the tenant ───────────
Write-Output "`n[Step 0] Discovering Linux VM SKUs across all subscriptions..."

$DiscoveredSkus     = [System.Collections.Generic.List[string]]::new()
$SubscriptionErrors = [System.Collections.Generic.List[string]]::new()

# Get all enabled subscriptions in the tenant
$AllSubscriptions = Get-AzSubscription | Where-Object { $_.State -eq "Enabled" }
Write-Output "  Enabled subscriptions found: $($AllSubscriptions.Count)"

# Save current context to restore after discovery
$OriginalContext = Get-AzContext

foreach ($Sub in $AllSubscriptions) {
    try {
        Write-Output "  Scanning: $($Sub.Name) ($($Sub.Id))"
        Set-AzContext -SubscriptionId $Sub.Id -TenantId $Sub.TenantId -Force | Out-Null

        # Get all VMs — filter to Linux OS only at source
        $LinuxVMs = Get-AzVM | Where-Object {
            $_.StorageProfile.OsDisk.OsType -eq "Linux"
        }

        if ($LinuxVMs -and $LinuxVMs.Count -gt 0) {
            Write-Output "    Linux VMs found: $($LinuxVMs.Count)"

            foreach ($VM in $LinuxVMs) {
                $Sku = $VM.HardwareProfile.VmSize
                if (-not [string]::IsNullOrWhiteSpace($Sku) -and
                    -not $DiscoveredSkus.Contains($Sku)) {
                    $DiscoveredSkus.Add($Sku)
                    Write-Output "    + New SKU discovered: $Sku"
                }
            }
        }
        else {
            Write-Output "    No Linux VMs found in this subscription."
        }
    }
    catch {
        $ErrMsg = "Failed to scan subscription '$($Sub.Name)': $_"
        Write-Warning "  $ErrMsg"
        $SubscriptionErrors.Add($ErrMsg)
    }
}

# Restore original context
if ($OriginalContext) {
    Set-AzContext -Context $OriginalContext -Force | Out-Null
}

# Sort for clean logging
$DiscoveredSkus = $DiscoveredSkus | Sort-Object

Write-Output "`n  ── Discovery Summary ──────────────────────────"
Write-Output "  Subscriptions scanned : $($AllSubscriptions.Count)"
Write-Output "  Subscriptions errored : $($SubscriptionErrors.Count)"
Write-Output "  Unique Linux SKUs found: $($DiscoveredSkus.Count)"
foreach ($Sku in $DiscoveredSkus) {
    Write-Output "    • $Sku"
}

# Safety guard — abort if no SKUs discovered
if ($DiscoveredSkus.Count -eq 0) {
    Write-Error "No Linux VM SKUs discovered across tenant. Aborting — table NOT truncated."
    exit 1
}

# ── Step 1: Fetch Linux pricing from API for all discovered SKUs ──────────
Write-Output "`n[Step 1] Fetching Linux prices from Azure Retail Prices API..."

$PricingResults = [System.Collections.Generic.List[object]]::new()

foreach ($Sku in $DiscoveredSkus) {
    Write-Output "  Querying: $Sku"
    $Result = Get-LinuxVmSkuPrice `
                -SkuName    $Sku `
                -Region     $Region `
                -ApiUrl     $PricingApiUrl `
                -ApiVersion $ApiVersion

    $PricingResults.Add($Result)

    if ($Result.Found) {
        Write-Output ("    ✔ {0,-30} → `${1}/hr  [Meter: {2}]" -f $Sku, $Result.PricePerHour, $Result.MeterName)
    }
    else {
        Write-Warning "    ✘ $Sku → No Linux price found in $Region"
    }
}

$FoundCount  = ($PricingResults | Where-Object {  $_.Found }).Count
$MissingSkus = ($PricingResults | Where-Object { -not $_.Found }).VmSkuName

Write-Output "`n  Prices fetched successfully : $FoundCount / $($DiscoveredSkus.Count)"
if ($MissingSkus) {
    Write-Warning "  SKUs with no price in $Region : $($MissingSkus -join ', ')"
}

# Safety guard — do not truncate if API returned nothing at all
if ($FoundCount -eq 0) {
    Write-Error "No Linux prices returned from API for any SKU. Aborting — table NOT truncated."
    exit 1
}

# ── Step 2: Open SQL connection ───────────────────────────────────────────
Write-Output "`n[Step 2] Connecting to SQL: $SQL_Server_Name / $SQL_DB_Name"

$SqlConnection = New-Object System.Data.SqlClient.SqlConnection
$SqlConnection.ConnectionString = "Server=$SQL_Server_Name;" +
                                  "Database=$SQL_DB_Name;" +
                                  "User ID=$($SQLServerCred.UserName);" +
                                  "Password=$($SQLServerCred.GetNetworkCredential().Password);" +
                                  "Encrypt=True;TrustServerCertificate=False;" +
                                  "Connection Timeout=30;"
try {
    $SqlConnection.Open()
    Write-Output "  Connection opened."
}
catch {
    Write-Error "SQL connection failed: $_"
    exit 1
}

# ── Step 3: Truncate ──────────────────────────────────────────────────────
Write-Output "`n[Step 3] Truncating dbo.AzureVmSkuPricing..."

$TruncateCmd             = $SqlConnection.CreateCommand()
$TruncateCmd.CommandText = "TRUNCATE TABLE dbo.AzureVmSkuPricing"

try {
    $TruncateCmd.ExecuteNonQuery() | Out-Null
    Write-Output "  Table truncated."
}
catch {
    Write-Error "TRUNCATE failed: $_"
    $SqlConnection.Close()
    exit 1
}

# ── Step 4: Insert fresh pricing rows ────────────────────────────────────
Write-Output "`n[Step 4] Inserting $FoundCount Linux pricing rows..."

$InsertCmd             = $SqlConnection.CreateCommand()
$InsertCmd.CommandText = @"
INSERT INTO dbo.AzureVmSkuPricing (VmSkuName, PricePerHour)
VALUES (@VmSkuName, @PricePerHour)
"@

$null = $InsertCmd.Parameters.Add("@VmSkuName",    [System.Data.SqlDbType]::NVarChar, 100)
$null = $InsertCmd.Parameters.Add("@PricePerHour", [System.Data.SqlDbType]::Decimal)
$InsertCmd.Parameters["@PricePerHour"].Precision = 10
$InsertCmd.Parameters["@PricePerHour"].Scale     = 6

$InsertedCount = 0
foreach ($Row in $PricingResults | Where-Object { $_.Found }) {
    $InsertCmd.Parameters["@VmSkuName"].Value    = $Row.VmSkuName
    $InsertCmd.Parameters["@PricePerHour"].Value = $Row.PricePerHour

    try {
        $InsertCmd.ExecuteNonQuery() | Out-Null
        $InsertedCount++
        Write-Output ("  Inserted: {0,-30} `${1}/hr" -f $Row.VmSkuName, $Row.PricePerHour)
    }
    catch {
        Write-Warning "  Insert failed for '$($Row.VmSkuName)': $_"
    }
}

$SqlConnection.Close()
Write-Output "`n  SQL connection closed."

#endregion

#region ── Summary ──────────────────────────────────────────────────────────

Write-Output "`n========================================"
Write-Output "REFRESH COMPLETE"
Write-Output "Timestamp          : $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') UTC"
Write-Output "OS Filter          : $OSType (no Windows licensing cost)"
Write-Output "Region             : $Region"
Write-Output "Subscriptions Scanned : $($AllSubscriptions.Count)"
Write-Output "Unique SKUs Discovered: $($DiscoveredSkus.Count)"
Write-Output "Rows Inserted      : $InsertedCount / $($DiscoveredSkus.Count)"
if ($MissingSkus) {
    Write-Output "No Price in Region : $($MissingSkus -join ', ')"
}
if ($SubscriptionErrors.Count -gt 0) {
    Write-Output "Subscription Errors: $($SubscriptionErrors.Count) — check warnings above"
}
Write-Output "========================================"

#endregion
