<#
.SYNOPSIS
    Monthly VM SKU pricing refresh — Linux compute price only (no OS licensing).
    Truncates AzureVmSkuPricing and repopulates from Azure Retail Prices API
    for West Europe region.

.SCHEDULE
    1st of every month at 00:30 UTC
#>

#region ── Configuration ────────────────────────────────────────────────────

$SQL_Server_Name = Get-AutomationVariable    -Name "SQL_Server_Name"
$SQL_DB_Name     = Get-AutomationVariable    -Name "SQL_DB_Name"
$SQLServerCred   = Get-AutomationPSCredential -Name "SQLServerCred"

$PricingApiUrl = "https://prices.azure.com/api/retail/prices"
$ApiVersion    = "2023-01-01-preview"
$Region        = "westeurope"
$OSType        = "Linux"   # Controls filtering — bare compute, no Windows licensing

# ✏️ ADD / REMOVE SKUs here as needed
$TargetSkus = @(
    "Standard_B2s",
    "Standard_B4ms",
    "Standard_D2s_v3",
    "Standard_D4s_v3",
    "Standard_D8s_v3",
    "Standard_D2s_v5",
    "Standard_D4s_v5",
    "Standard_E4s_v3",
    "Standard_E8s_v3",
    "Standard_F4s_v2",
    "Standard_F8s_v2"
)

#endregion

#region ── Helper: Fetch Linux-only price for a single SKU ──────────────────

function Get-LinuxVmSkuPrice {
    param (
        [string]$SkuName,
        [string]$Region,
        [string]$ApiUrl,
        [string]$ApiVersion
    )

    # API-level filter:
    # - armSkuName       → exact SKU match
    # - armRegionName    → West Europe
    # - priceType        → Consumption (Pay-As-You-Go, not Reservation)
    # - serviceName      → Virtual Machines only
    # - not Windows      → exclude Windows OS licensed meters at source
    $Filter = "armSkuName eq '$SkuName' " +
              "and armRegionName eq '$Region' " +
              "and priceType eq 'Consumption' " +
              "and serviceName eq 'Virtual Machines' " +
              "and not contains(meterName, 'Windows') " +
              "and not contains(skuName, 'Windows')"

    $Uri = "$ApiUrl`?api-version=$ApiVersion&`$filter=$([Uri]::EscapeDataString($Filter))"

    try {
        $Response = Invoke-RestMethod -Uri $Uri -Method GET -ContentType "application/json"

        if ($Response.Items -and $Response.Items.Count -gt 0) {

            # Local safety filter — belt and braces:
            # Keep only rows where skuName has NO OS suffix
            # Linux rows appear as e.g. "D4s v3", "D4s v5"
            # Windows rows appear as e.g. "D4s v3 Windows", "D4s v5 Windows"
            $LinuxItem = $Response.Items | Where-Object {
                $_.skuName        -notmatch "Windows|Low Priority|Spot" -and
                $_.meterName      -notmatch "Windows|Low Priority|Spot" -and
                $_.unitOfMeasure  -eq "1 Hour"
            } | Select-Object -First 1

            if ($LinuxItem) {
                return [PSCustomObject]@{
                    VmSkuName    = $SkuName
                    PricePerHour = [decimal]$LinuxItem.retailPrice
                    MeterName    = $LinuxItem.meterName
                    SkuNameRaw   = $LinuxItem.skuName
                    Found        = $true
                }
            }

            Write-Warning "  API returned results for '$SkuName' but none matched Linux/standard filter."
        }
        else {
            Write-Warning "  No results returned from API for SKU: $SkuName"
        }

        return [PSCustomObject]@{
            VmSkuName    = $SkuName
            PricePerHour = 0
            MeterName    = $null
            SkuNameRaw   = $null
            Found        = $false
        }
    }
    catch {
        Write-Warning "  API call failed for SKU '$SkuName': $_"
        return [PSCustomObject]@{
            VmSkuName    = $SkuName
            PricePerHour = 0
            MeterName    = $null
            SkuNameRaw   = $null
            Found        = $false
        }
    }
}

#endregion

#region ── Main ─────────────────────────────────────────────────────────────

Write-Output "========================================"
Write-Output "VM SKU Pricing Refresh ($OSType Only)"
Write-Output "Timestamp : $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') UTC"
Write-Output "Region    : $Region"
Write-Output "SKU Count : $($TargetSkus.Count)"
Write-Output "========================================"

# ── Step 1: Fetch Linux pricing from API ─────────────────────────────────
Write-Output "`n[Step 1] Fetching Linux compute prices from Azure Retail Prices API..."

$PricingResults = [System.Collections.Generic.List[object]]::new()

foreach ($Sku in $TargetSkus) {
    Write-Output "  Querying: $Sku"
    $Result = Get-LinuxVmSkuPrice `
                -SkuName    $Sku `
                -Region     $Region `
                -ApiUrl     $PricingApiUrl `
                -ApiVersion $ApiVersion

    $PricingResults.Add($Result)

    if ($Result.Found) {
        Write-Output ("    ✔ {0,-25} → `${1}/hr  [Meter: {2}]" -f $Sku, $Result.PricePerHour, $Result.MeterName)
    } else {
        Write-Warning "    ✘ $Sku → Not found"
    }
}

$FoundCount  = ($PricingResults | Where-Object { $_.Found }).Count
$MissingSkus = ($PricingResults | Where-Object { -not $_.Found }).VmSkuName

Write-Output "`n  Prices fetched successfully : $FoundCount / $($TargetSkus.Count)"
if ($MissingSkus) {
    Write-Warning "  SKUs with no price returned : $($MissingSkus -join ', ')"
}

# Safety guard — do not truncate if API returned nothing
if ($FoundCount -eq 0) {
    Write-Error "No Linux prices returned from API. Aborting — table NOT truncated."
    exit 1
}

# ── Step 2: Open SQL connection ───────────────────────────────────────────
Write-Output "`n[Step 2] Connecting to SQL: $SQL_Server_Name / $SQL_DB_Name"

$SqlConnection = New-Object System.Data.SqlClient.SqlConnection
$SqlConnection.ConnectionString = "Server=$SQL_Server_Name;" +
                                  "Database=$SQL_DB_Name;" +
                                  "User ID=$($SQLServerCred.UserName);" +
