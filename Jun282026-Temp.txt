if ($Subscriptions) {
    foreach ($Subscription in $Subscriptions) {
        Write-Verbose "Changing to Subscription $($Subscription.Name)" -Verbose

        $Context  = Set-AzContext -TenantId $Subscription.TenantId -SubscriptionId $Subscription.Id -Force
        $Name     = $Subscription.Name
        $TenantId = $Subscription.TenantId
        $SubId    = $Subscription.Id
        $Date     = Get-Date -Format "yyyy-MM-dd"

        $Disks = Get-AzDisk | Where-Object { $_.DiskState -eq "Unattached" -and $null -eq $_.ManagedBy }

        if (-not $Disks -or $Disks.Count -eq 0) {
            Write-Host "No unattached disks found in subscription '$Name'" -ForegroundColor Cyan
            continue
        }

        Write-Host "We found a count of $($Disks.Count) unattached disks in subscription '$Name'" -ForegroundColor Yellow
        Write-Host "Getting usage details for all resources in this subscription"

        $All_Az_Consumption_Usage_Details = Get-AzConsumptionUsageDetail

        Write-Host "Building usage lookup"
        $usageLookup = @{}
        foreach ($item in $All_Az_Consumption_Usage_Details) {
            if (-not [string]::IsNullOrWhiteSpace($item.InstanceId)) {
                if (-not $usageLookup.ContainsKey($item.InstanceId)) {
                    $usageLookup[$item.InstanceId] = [PSCustomObject]@{
                        PreTaxCost = 0.0
                        Currency   = $item.Currency
                    }
                }

                $usageLookup[$item.InstanceId].PreTaxCost += [double]($item.PretaxCost ?? 0)
                if (-not $usageLookup[$item.InstanceId].Currency -and $item.Currency) {
                    $usageLookup[$item.InstanceId].Currency = $item.Currency
                }
            }
        }

        Write-Host "Preparing output objects"
        $results = foreach ($Disk in $Disks) {
            Write-Host "Processing disk '$($Disk.Name)'"

            $costInfo = $usageLookup[$Disk.Id]
            $preTax   = if ($costInfo) { $costInfo.PreTaxCost } else { 0 }
            $currency = if ($costInfo) { $costInfo.Currency } else { $null }

            [PSCustomObject]@{
                DiskName              = $Disk.Name
                ResourceGroup         = $Disk.ResourceGroupName
                Subscription          = $Name
                SubscriptionID        = $SubId
                ResourceId            = $Disk.Id
                Location              = $Disk.Location
                DiskSize              = $Disk.DiskSizeGB
                Disk_Cost_PreTaxCost  = $preTax
                Disk_Cost_Currency    = $currency
            }
        }

        $FileName = "OrphanedDisks-$Name-$Date.csv"
        $CsvPath  = Join-Path $env:TEMP $FileName
        $results | Export-Csv $CsvPath -NoTypeInformation

        Write-Host "Writing results to SQL"

        $sqlConnection = New-Object System.Data.SqlClient.SqlConnection
        $sqlConnection.ConnectionString = "Server=$SQL_Server_Name;Database=$SQL_DB_Name;User ID=$($SQLServerCred.UserName);Password=$($SQLServerCred.GetNetworkCredential().Password);Encrypt=True;TrustServerCertificate=False;"
        $sqlConnection.Open()

        $sqlCmd = $sqlConnection.CreateCommand()
        $sqlCmd.CommandText = @"
INSERT INTO Azure_OrphanedDisks_Logs
(Disk, ResourceId, RG, Subscription, SubscriptionID, DiskSize, Location, Disk_Cost_PreTaxCost, Disk_Cost_Currency, [Date])
VALUES
(@Disk, @ResourceId, @RG, @Subscription, @SubscriptionID, @DiskSize, @Location, @DiskCost, @Currency, GETDATE())
"@

        $null = $sqlCmd.Parameters.Add("@Disk", [System.Data.SqlDbType]::NVarChar, 255)
        $null = $sqlCmd.Parameters.Add("@ResourceId", [System.Data.SqlDbType]::NVarChar, 500)
        $null = $sqlCmd.Parameters.Add("@RG", [System.Data.SqlDbType]::NVarChar, 255)
        $null = $sqlCmd.Parameters.Add("@Subscription", [System.Data.SqlDbType]::NVarChar, 255)
        $null = $sqlCmd.Parameters.Add("@SubscriptionID", [System.Data.SqlDbType]::NVarChar, 100)
        $null = $sqlCmd.Parameters.Add("@DiskSize", [System.Data.SqlDbType]::Int)
        $null = $sqlCmd.Parameters.Add("@Location", [System.Data.SqlDbType]::NVarChar, 100)
        $null = $sqlCmd.Parameters.Add("@DiskCost", [System.Data.SqlDbType]::Decimal)
        $null = $sqlCmd.Parameters["@DiskCost"].Precision = 18
        $null = $sqlCmd.Parameters["@DiskCost"].Scale = 4
        $null = $sqlCmd.Parameters.Add("@Currency", [System.Data.SqlDbType]::NVarChar, 20)

        foreach ($row in $results) {
            $sqlCmd.Parameters["@Disk"].Value           = $row.DiskName
            $sqlCmd.Parameters["@ResourceId"].Value     = $row.ResourceId
            $sqlCmd.Parameters["@RG"].Value             = $row.ResourceGroup
            $sqlCmd.Parameters["@Subscription"].Value   = $row.Subscription
            $sqlCmd.Parameters["@SubscriptionID"].Value = $row.SubscriptionID
            $sqlCmd.Parameters["@DiskSize"].Value       = $row.DiskSize
            $sqlCmd.Parameters["@Location"].Value       = $row.Location
            $sqlCmd.Parameters["@DiskCost"].Value       = [decimal]$row.Disk_Cost_PreTaxCost
            $sqlCmd.Parameters["@Currency"].Value       = if ($row.Disk_Cost_Currency) { $row.Disk_Cost_Currency } else { [DBNull]::Value }

            $sqlCmd.ExecuteNonQuery() | Out-Null
        }

        $sqlConnection.Close()

Write-Host "Writing results to SQL"

$sqlConnection = New-Object System.Data.SqlClient.SqlConnection
$sqlConnection.ConnectionString = "Server=$SQL_Server_Name;Database=$SQL_DB_Name;User ID=$($SQLServerCred.UserName);Password=$($SQLServerCred.GetNetworkCredential().Password);Encrypt=True;TrustServerCertificate=False;"
$sqlConnection.Open()

try {
    #
    # 1. Insert orphaned disk rows
    #
    $insertCmd = $sqlConnection.CreateCommand()
    $insertCmd.CommandText = @"
INSERT INTO Azure_OrphanedDisks_Logs
(
    Disk,
    ResourceId,
    RG,
    Subscription,
    SubscriptionID,
    DiskSize,
    Location,
    Disk_Cost_PreTaxCost,
    Disk_Cost_Currency,
    [Date]
)
VALUES
(
    @Disk,
    @ResourceId,
    @RG,
    @Subscription,
    @SubscriptionID,
    @DiskSize,
    @Location,
    @DiskCost,
    @Currency,
    GETDATE()
)
"@
    $insertCmd.CommandType = [System.Data.CommandType]::Text

    [void]$insertCmd.Parameters.Add("@Disk", [System.Data.SqlDbType]::NVarChar, 255)
    [void]$insertCmd.Parameters.Add("@ResourceId", [System.Data.SqlDbType]::NVarChar, 500)
    [void]$insertCmd.Parameters.Add("@RG", [System.Data.SqlDbType]::NVarChar, 255)
    [void]$insertCmd.Parameters.Add("@Subscription", [System.Data.SqlDbType]::NVarChar, 255)
    [void]$insertCmd.Parameters.Add("@SubscriptionID", [System.Data.SqlDbType]::NVarChar, 100)
    [void]$insertCmd.Parameters.Add("@DiskSize", [System.Data.SqlDbType]::Int)
    [void]$insertCmd.Parameters.Add("@Location", [System.Data.SqlDbType]::NVarChar, 100)

    [void]$insertCmd.Parameters.Add("@DiskCost", [System.Data.SqlDbType]::Decimal)
    $insertCmd.Parameters["@DiskCost"].Precision = 18
    $insertCmd.Parameters["@DiskCost"].Scale = 4

    [void]$insertCmd.Parameters.Add("@Currency", [System.Data.SqlDbType]::NVarChar, 20)

    foreach ($row in $results) {
        $insertCmd.Parameters["@Disk"].Value           = $row.DiskName
        $insertCmd.Parameters["@ResourceId"].Value     = $row.ResourceId
        $insertCmd.Parameters["@RG"].Value             = $row.ResourceGroup
        $insertCmd.Parameters["@Subscription"].Value   = $row.Subscription
        $insertCmd.Parameters["@SubscriptionID"].Value = $row.SubscriptionID
        $insertCmd.Parameters["@DiskSize"].Value       = [int]$row.DiskSize
        $insertCmd.Parameters["@Location"].Value       = $row.Location
        $insertCmd.Parameters["@DiskCost"].Value       = [decimal]$row.Disk_Cost_PreTaxCost

        if ([string]::IsNullOrWhiteSpace($row.Disk_Cost_Currency)) {
            $insertCmd.Parameters["@Currency"].Value = [DBNull]::Value
        }
        else {
            $insertCmd.Parameters["@Currency"].Value = $row.Disk_Cost_Currency
        }

        [void]$insertCmd.ExecuteNonQuery()
    }

    Write-Host "Disk rows inserted successfully for subscription '$Name'" -ForegroundColor Green

    #
    # 2. Call stored procedure to get the sum for same subscription and same date
    #
    $spCmd = $sqlConnection.CreateCommand()
    $spCmd.CommandText = "dbo.usp_GetOrphanedDiskCostBySubscriptionDate"
    $spCmd.CommandType = [System.Data.CommandType]::StoredProcedure

    [void]$spCmd.Parameters.Add("@SubscriptionID", [System.Data.SqlDbType]::NVarChar, 100)
    $spCmd.Parameters["@SubscriptionID"].Value = $SubId

    [void]$spCmd.Parameters.Add("@TargetDate", [System.Data.SqlDbType]::Date)
    $spCmd.Parameters["@TargetDate"].Value = [datetime]::ParseExact($Date, "yyyy-MM-dd", $null)

    $reader = $spCmd.ExecuteReader()
    $spTable = New-Object System.Data.DataTable
    $spTable.Load($reader)
    $reader.Close()

    $totalDiskCostForSubscription = [decimal]0

    if ($spTable.Rows.Count -gt 0) {
        $firstRow = $spTable.Rows[0]

        if ($spTable.Columns.Contains("TotalDiskCost") -and $firstRow["TotalDiskCost"] -ne [DBNull]::Value) {
            $totalDiskCostForSubscription = [decimal]$firstRow["TotalDiskCost"]
        }
    }

    Write-Host "Total orphaned disk cost for subscription '$Name' on $Date is $totalDiskCostForSubscription" -ForegroundColor Cyan

    #
    # 3. Example email body usage
    #
    $emailBody = @"
Hello Team,

Subscription Name : $Name
Subscription ID   : $SubId
Run Date          : $Date
Total Orphaned Disk Cost : $totalDiskCostForSubscription

Regards,
Azure Automation
"@

    Write-Host $emailBody
}
finally {
    if ($null -ne $sqlConnection -and $sqlConnection.State -eq 'Open') {
        $sqlConnection.Close()
    }
}
