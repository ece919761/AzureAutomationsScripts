$Tags = @()
$mgGroups = Get-AzManagementGroup


foreach ($mg in $mgGroups) {
    $groupId = $mg.Id
    $groupName = $mg.Name

    try {
        $subscriptions = Get-AzManagementGroupSubscription -GroupName $groupName

        foreach ($sub in $subscriptions) {
            # Extract Subscription ID from the full resource ID
            $subId = ($sub.Id -split "/")[-1]
            Set-AzContext -TenantId "f66b6bd3-ebc2-4f54-8769-d22858de97c5" -SubscriptionId $subId-Force
            $resourceId = "/subscriptions/$subId"
            $subscriptionDetails = Get-AzTag -ResourceId $resourceId

            $tagObject = [ordered]@{
                ManagementGroupName = $groupName
                SubscriptionName    = $sub.DisplayName
                SubscriptionId      = $subId
                SubscriptionState   = $sub.State
                TenantId            = $sub.Tenant
            }
            if ($subscriptionDetails.Properties.TagsProperty) {
                foreach ($tagKey in $subscriptionDetails.Properties.TagsProperty.Keys) {
                    $tagValue = $subscriptionDetails.Properties.TagsProperty[$tagKey]
                    $tagObject[$tagKey] = $tagValue
                }
            }
            $Tags += New-Object PSObject -Property $tagObject
        }
    } catch {
        Write-Output "No subscriptions found or error for management group '$groupName'"
    }
}
