cd "lessons-learned-mcp-handoff 1"

# 1. Log in and target the right subscription
az login
az account set --subscription "GBGKrishnaTest"

# 2. Validate first (no changes made)
az deployment sub what-if \
  --location westeurope \
  --template-file infra/main.bicep \
  --parameters infra/params/prod.private.bicepparam

# 3. Deploy
az deployment sub create \
  --name lessons-learned-mcp-prod \
  --location westeurope \
  --template-file infra/main.bicep \
  --parameters infra/params/prod.private.bicepparam

# 4. Get outputs (endpoint URL, Key Vault name, etc.)
az deployment sub show --name lessons-learned-mcp-prod --query properties.outputs
